### Worked Example — Sofa:
```python
import cadquery as cq
import math

# ── MASTER PARAMETERS ──────────────────────────────────────────
seat_length    = 2000.0  # mm — three-seater width
seat_depth     = 900.0   # mm — front-to-back
leg_height     = 200.0   # mm — floor to seat underside
cushion_t      = 120.0   # mm — EN 1335 upholstered seat pad
back_h         = 650.0   # mm — backrest height above seat top
back_depth     = 130.0   # mm — backrest thickness
arm_w          = 120.0   # mm — armrest width (X)
arm_h          = 300.0   # mm — armrest height above seat bottom

# ── DERIVED VALUES ─────────────────────────────────────────────
leg_dia = max(leg_height / 10.0, 40.0)  # structural: H/10, min 40mm
leg_inset_x = leg_dia / 2.0 + 30.0
leg_inset_y = leg_dia / 2.0 + 30.0
seat_top_z  = leg_height + cushion_t
back_y      = -(seat_depth / 2.0 - back_depth / 2.0)  # flush to rear

# ── VALIDATION ─────────────────────────────────────────────────
assert leg_height >= 150, "Sofa seat too low (<150mm)"
assert back_h / back_depth <= 10.0, "Backrest too thin for height (H/t>10)"

# ── GEOMETRY ──────────────────────────────────────────────────
leg_pts = [
    ( seat_length/2 - leg_inset_x,  seat_depth/2 - leg_inset_y),
    (-seat_length/2 + leg_inset_x,  seat_depth/2 - leg_inset_y),
    ( seat_length/2 - leg_inset_x, -seat_depth/2 + leg_inset_y),
    (-seat_length/2 + leg_inset_x, -seat_depth/2 + leg_inset_y),
]
legs = cq.Workplane("XY").pushPoints(leg_pts).circle(leg_dia/2).extrude(leg_height)
seat = (cq.Workplane("XY")
        .box(seat_length, seat_depth, cushion_t, centered=(True, True, False))
        .translate((0, 0, leg_height)))
backrest = (cq.Workplane("XY")
            .box(seat_length, back_depth, back_h, centered=(True, True, False))
            .translate((0, back_y, seat_top_z)))
left_arm  = (cq.Workplane("XY")
             .box(arm_w, seat_depth, arm_h, centered=(True, True, False))
             .translate(( seat_length/2 - arm_w/2, 0, leg_height)))
right_arm = (cq.Workplane("XY")
             .box(arm_w, seat_depth, arm_h, centered=(True, True, False))
             .translate((-seat_length/2 + arm_w/2, 0, leg_height)))
result = seat.union(backrest).union(legs).union(left_arm).union(right_arm)
```

### Worked Example — Bolt Flange (pressure vessel):
```python
import cadquery as cq
import math

bore_dia      = 50.0    # mm
pressure_mpa  =  8.0    # MPa
sigma_allow   = 250.0   # MPa — carbon steel yield
safety_factor =  2.5    # ASME VIII minimum

wall_t        = pressure_mpa * bore_dia / (2 * sigma_allow / safety_factor)  # Barlow
wall_t        = max(wall_t, bore_dia / 20)
flange_od     = bore_dia + 6 * wall_t
bolt_circle_r = 0.72 * flange_od / 2  # ASME B16.5
bolt_dia      = max(6.0, 0.8 * wall_t)
bolt_count    = 8

assert wall_t >= bore_dia / 20, f"Wall {wall_t:.2f}mm too thin"

hub    = cq.Workplane("XY").circle(bore_dia/2 + wall_t).extrude(40)
flange = cq.Workplane("XY").circle(flange_od/2).extrude(15)
bore   = cq.Workplane("XY").circle(bore_dia/2).extrude(55)
bolt_pts = [(bolt_circle_r * math.cos(2*math.pi*i/bolt_count),
             bolt_circle_r * math.sin(2*math.pi*i/bolt_count)) for i in range(bolt_count)]
result = (hub.union(flange).cut(bore)
          .faces(">Z").workplane().pushPoints(bolt_pts).hole(bolt_dia))
```
