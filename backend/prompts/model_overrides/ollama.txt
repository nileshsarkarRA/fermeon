```python
import cadquery as cq
import math

# ── MASTER PARAMETERS ────────────────────────
seat_length  = 2000.0  # mm — three-seater total width
seat_depth   =  900.0  # mm — front-to-back
leg_height   =  200.0  # mm — floor to seat bottom
cushion_t    =  120.0  # mm — EN 1335 seat pad
back_h       =  650.0  # mm — backrest height above seat
back_depth   =  130.0  # mm — backrest thickness
arm_w        =  120.0  # mm — armrest width
arm_h        =  300.0  # mm — armrest height above seat bottom

# ── DERIVED VALUES ────────────────────────────
leg_dia     = max(leg_height / 10.0, 40.0)  # structural: H/10, min 40mm
leg_inset_x = leg_dia / 2.0 + 30.0
leg_inset_y = leg_dia / 2.0 + 30.0
seat_top_z  = leg_height + cushion_t        # Z=320
back_y      = -(seat_depth/2.0 - back_depth/2.0)  # flush to rear

# ── ASSERTIONS ──────────────────────────────
assert leg_height >= 150.0, "Seat too low"
assert back_h / back_depth <= 10.0, "Backrest too thin"

# ── GEOMETRY ───────────────────────────────
leg_pts = [
    ( seat_length/2 - leg_inset_x,  seat_depth/2 - leg_inset_y),
    (-seat_length/2 + leg_inset_x,  seat_depth/2 - leg_inset_y),
    ( seat_length/2 - leg_inset_x, -seat_depth/2 + leg_inset_y),
    (-seat_length/2 + leg_inset_x, -seat_depth/2 + leg_inset_y),
]
legs = cq.Workplane("XY").pushPoints(leg_pts).circle(leg_dia/2).extrude(leg_height)
seat = (cq.Workplane("XY")
        .box(seat_length, seat_depth, cushion_t, centered=(True, True, False))
        .translate((0, 0, leg_height)))
backrest = (cq.Workplane("XY")
            .box(seat_length, back_depth, back_h, centered=(True, True, False))
            .translate((0, back_y, seat_top_z)))
left_arm  = (cq.Workplane("XY")
             .box(arm_w, seat_depth, arm_h, centered=(True, True, False))
             .translate(( seat_length/2 - arm_w/2, 0, leg_height)))
right_arm = (cq.Workplane("XY")
             .box(arm_w, seat_depth, arm_h, centered=(True, True, False))
             .translate((-seat_length/2 + arm_w/2, 0, leg_height)))
result = seat.union(backrest).union(legs).union(left_arm).union(right_arm)
```
