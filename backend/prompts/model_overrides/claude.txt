ASSISTANT:

```python
import cadquery as cq
import math

# ── MASTER PARAMETERS ─────────────────────────────────────────
bore_dia     = 50.0   # mm
P_mpa        =  8.0   # MPa design pressure
sigma_y      = 250.0  # MPa carbon steel yield
SF           =  2.5   # ASME VIII safety factor
flange_h     = 20.0   # mm flange thickness
hub_h        = 40.0   # mm hub height

# ── DERIVED VALUES (Barlow + ASME B16.5) ───────────────────────
wall_t       = P_mpa * (bore_dia/2) / (sigma_y/SF)         # ASME thin-wall
wall_t       = max(wall_t, bore_dia / 20)                   # D/20 practical minimum
hub_od       = bore_dia + 2 * wall_t
flange_od    = bore_dia + 6 * wall_t                        # 3× wall each side
bolt_r       = 0.72 * flange_od / 2                        # ASME B16.5
bolt_dia     = max(6.0, 0.8 * wall_t)
bolt_n       = 8

# ── ASSERTIONS ────────────────────────────────────────────────
assert wall_t >= bore_dia / 20, f"Wall {wall_t:.1f} too thin"
assert bolt_dia >= 6.0, "Minimum M6 bolt"

# ── GEOMETRY ──────────────────────────────────────────────────
hub    = cq.Workplane("XY").circle(hub_od/2).extrude(hub_h)
flange = cq.Workplane("XY").circle(flange_od/2).extrude(flange_h)
boring = cq.Workplane("XY").circle(bore_dia/2).extrude(hub_h + flange_h)
bolt_pts = [(bolt_r*math.cos(2*math.pi*i/bolt_n),
             bolt_r*math.sin(2*math.pi*i/bolt_n)) for i in range(bolt_n)]
result = (hub.union(flange).cut(boring)
          .faces(">Z").workplane().pushPoints(bolt_pts).hole(bolt_dia))
```

ASSISTANT:
```python
import cadquery as cq
import math

# ── MASTER PARAMETERS ─────────────────────────────────────────
base_l = 150.0; base_w = 80.0; wall_t = 6.0   # mm
vert_h = 100.0                                  # mm vertical wall height
gusset_l = 60.0                                 # mm gusset horizontal span
bolt_d   = 6.6                                  # mm M6 clearance [ISO 273]

# ── DERIVED VALUES ───────────────────────────────────────────
gusset_t   = wall_t * 0.6                       # rib: 60% of wall thickness
bolt_inset = bolt_d / 2 + 5.0

# ── GEOMETRY ─────────────────────────────────────────────────
base = cq.Workplane("XY").box(base_l, base_w, wall_t, centered=(True,True,False))
wall = (cq.Workplane("XY")
        .box(wall_t, base_w, vert_h, centered=(True,True,False))
        .translate((-base_l/2 + wall_t/2, 0, wall_t)))
gusset = (cq.Workplane("XZ")
          .polygon(3, gusset_l).extrude(gusset_t)
          .translate((-base_l/2 + wall_t, 0, wall_t)))
bolt_pts = [(base_l/2 - bolt_inset, base_w/2 - bolt_inset),
            (base_l/2 - bolt_inset, -base_w/2 + bolt_inset)]
result = (base.union(wall).union(gusset)
          .faces(">Z").workplane().pushPoints(bolt_pts).hole(bolt_d))
```
